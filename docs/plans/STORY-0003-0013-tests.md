function extractSection(content: string, heading: string): string {
  const start = content.indexOf(heading);
  if (start === -1) return "";
  const afterHeading = content.indexOf("\n## ", start + 1);
  return content.slice(
    start + heading.length,
    afterHeading > -1 ? afterHeading : undefined,
  ).trim();
}

describe("source template TDD section consistency", () => {
  it.each([
    ["## TDD Commit Format"],
    ["## Atomic TDD Commit Rules"],

    (heading) => {
      const claudeSection = extractSection(claudeContent, heading);

## 11. Risk Areas

| Risk | Likelihood | Impact | Mitigation |
|------|-----------|--------|------------|
| Golden files not regenerated before commit | High | High | CI fails immediately -- `pipelineMatchesGoldenFiles` assertion detects content mismatch and reports unified diff |
| TDD sections added at wrong insertion point | Medium | Medium | Section ordering test validates that TDD sections appear between `## Tagging Releases` and `## Integration Notes` |
| Claude and GitHub templates have different TDD content | Medium | High | Source template consistency tests (G6b) compare extracted TDD sections byte-for-byte between both templates |
| `.agents/` copy diverges from `.claude/` copy | Low | High | Dual copy tests (G6a) compare all 8 profiles. Also caught by byte-for-byte tests transitively |
| Existing Conventional Commits content accidentally modified | Low | High | Backward compatibility tests (G5) verify all 8 commit types and 5 existing sections are still present |
| Content language violation (RULE-012) | Low | Minor | All new content must be written in English. Content tests use English phrases for assertions |
| Template placeholder issues in new sections | None | None | New sections contain no profile-specific placeholders. They are static Markdown content that passes through `TemplateEngine` unchanged |
| Content assertions too brittle | Medium | Medium | Use `toContain()` for headers and key phrases, `toMatch()` with case-insensitive regex for flexible assertions. Avoid exact line matching |
| Test file discovery by Vitest | Low | Low | New test file at `tests/node/content/` must match the glob pattern in `vitest.config.ts` (`tests/**/*.test.ts`). Previous story already established the `tests/node/content/` directory pattern |
